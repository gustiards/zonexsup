<script>
const WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbwcMaIahM5suFWQhZ4e1MVGptmfEmzlB5UjogD_-oUebMe8RzYVQKU5S7x4JN9HjZB8qA/exec';
const SPREADSHEET_URL = 'https://docs.google.com/spreadsheets/d/1seFxyBxAhPmGLHcJBefeu4Ror-yeKA4P/edit';

let historyPage = ['home'];

function openPage(page) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById(page).classList.add('active');
  if (historyPage.at(-1) !== page) historyPage.push(page);
}

function goBack() {
  historyPage.pop();
  openPage(historyPage.pop() || 'home');
}

function openSheet() {
  window.open(SPREADSHEET_URL, '_blank');
}

function showLoading() {
  document.getElementById('loadingOverlay').classList.add('active');
}
function hideLoading() {
  document.getElementById('loadingOverlay').classList.remove('active');
}

function submitData(btn) {
  if (!zone.value.trim()) {
    alert('ZONE wajib diisi');
    return;
  }

  const page = document.querySelector('.page.active').id;
  let payload = { zone: zone.value.trim() };

  if (page === 'firstCounting') {
    payload.fc_counter = fc_counter.value;
    payload.fc_time_in = fc_time_in.value;
    payload.fc_time_out = fc_time_out.value;
    payload.fc_checked = fc_checked.value;
  }

  if (page === 'scanning') {
    payload.scan_name = scan_name.value;
    payload.scan_time_in = scan_time_in.value;
    payload.scan_time_out = scan_time_out.value;
  }

  btn.disabled = true;
  showLoading();

  fetch(WEB_APP_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  })
  .then(r => r.json())
  .then(r => {
    hideLoading();
    btn.disabled = false;
    alert(r.message);
    if (r.success) openPage('home');
  })
  .catch(() => {
    hideLoading();
    btn.disabled = false;
    alert('Gagal terhubung ke server');
  });
}

if ('serviceWorker' in navigator) {
  window.addEventListener('load', () => {
    navigator.serviceWorker.register('/sw.js');
  });
}
</script>
